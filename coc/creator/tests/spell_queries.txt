# Spells

## Retrieve spells based on its attributes



### Retrive all spells and obtain its uuid

query{
  allSpells{
    edges{
      node{
     	 uuid
      }
    }
  }
}


### Retrieve a node by a specific uuid and obtain its name

query{
  allSpells(uuid: "748a203b-1796-4084-8816-260ee0c58a74"){
    edges{
      node{
         name
      }
    }
  }
}

### Update a spell

mutation{
  spellMutate(
    input:{
    	uuid: "748a203b-1796-4084-8816-260ee0c58a74",
    	method: "UPDATE",
    	notes: "test",
      user: 1
    }){
    spell{
      notes
    }
  }
}

### Delete a spell

mutation{
  spellMutate(
    input:{
      uuid: "748a203b-1796-4084-8816-260ee0c58a74",
    	method: "DELETE",
      user: 1
    }){
    spell{
        uuid
    }
  }
}


### Create a spell

mutation{
  spellMutate(
    input:{
    	method: "CREATE",
    	notes: "test",
      name: "tester spell",
      description:"We may ommit this",
      cost: "1D6 Sanity",
      castingTime: "Immediate",
      user: 1
    }){
    spell{
      uuid,
      notes,
      name
    }
  }
}
