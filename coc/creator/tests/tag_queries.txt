# Tags

## Retrieve all tags but obtain only the uuid for each one of them.

query{
  allTags{
    edges{
      node{
        uuid
      }
    }
  }
}


## Retrieve one tag by the node uuid

query{
	allTags(uuid:"6c569beb-c4d6-4b82-9f68-1c257401514c"){
  	edges{
      node{
        title,
        id,
  	  }
    }
  }
}

## Search tag by node id
> Notice the id we have for the elements is different to the id the node has
> this is because the node model wraps around our model given that we use
> relays, maybe in the future this will be different.

query{
  tag(id: "VGFnTm9kZToy"){
    title
  }
}

## Create a tag test

mutation{
  tagMutate(
    input:{
      method: "CREATE",
      title: "test-tag",
      user: 1
    }){
      tag{
        title
      }
  }
}

## Edit tag
> In this scenario we need the uuid to find the tag we want to update
> and we pass the new title that we want to have at our tag

mutation{
  tagMutate(
      input:{
        method: "UPDATE",
        user: 1,
        title: "test-tag2",
        uuid: "ae42d667-b762-4818-a479-1d5abe0620c1"
   }){
    tag{
    id,
    uuid,
    title
    }
   }
}

## Delete a tag
> In this scenario we pass the method which is DEL that stands for DELETE
> we need to pass the uuid as well so the tag can be find and later deleted.

mutation{
  tagMutate(
      input: {
        user: 1,
        uuid: "9cff8439-acaa-4503-bc19-7c05affa9d0f",
        method: "DELETE"
      }){
    tag{
      id,
      uuid,
      title
    }
  }
}
